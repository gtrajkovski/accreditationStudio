## JavaScript

File: `static/js/bulk_remediation.js`

```javascript
const BulkRemediation = {
    institutionId: null,
    currentJobId: null,
    eventSource: null,

    init(institutionId) {
        this.institutionId = institutionId;
        this.bindEvents();
        this.updatePreview();
    },

    bindEvents() {
        // Scope type radio buttons
        document.querySelectorAll('input[name="scope_type"]').forEach(radio => {
            radio.addEventListener('change', () => this.onScopeChange());
        });

        // Start button
        document.getElementById('start-btn').addEventListener('click', () => this.startJob());

        // Pause/Cancel buttons
        document.getElementById('pause-btn').addEventListener('click', () => this.pauseJob());
        document.getElementById('cancel-btn').addEventListener('click', () => this.cancelJob());

        // Approval buttons
        document.getElementById('approve-all-btn').addEventListener('click', () => this.approveAll());
    },

    onScopeChange() {
        const scopeType = document.querySelector('input[name="scope_type"]:checked').value;

        // Enable/disable selects based on scope type
        document.getElementById('doc-type-select').disabled = scopeType !== 'doc_type';
        document.getElementById('program-select').disabled = scopeType !== 'program';
        document.querySelector('.severity-checks').style.display =
            scopeType === 'severity' ? 'block' : 'none';

        this.updatePreview();
    },

    async updatePreview() {
        const scope = this.getScope();

        const response = await fetch(`/api/institutions/${this.institutionId}/bulk-remediation/preview`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(scope)
        });

        const data = await response.json();

        document.getElementById('preview-docs').textContent = data.document_count;
        document.getElementById('preview-findings').textContent = data.total_findings;
        document.getElementById('start-btn').disabled = data.document_count === 0;
    },

    getScope() {
        const scopeType = document.querySelector('input[name="scope_type"]:checked').value;
        const scope = { scope_type: scopeType };

        if (scopeType === 'doc_type') {
            const selected = document.getElementById('doc-type-select').value;
            if (selected) scope.doc_types = [selected];
        } else if (scopeType === 'program') {
            const selected = document.getElementById('program-select').value;
            if (selected) scope.program_ids = [selected];
        } else if (scopeType === 'severity') {
            scope.severities = Array.from(
                document.querySelectorAll('.severity-checks input:checked')
            ).map(cb => cb.value);
        }

        return scope;
    },

    async startJob() {
        const scope = this.getScope();

        // Create job
        const createResponse = await fetch(`/api/institutions/${this.institutionId}/bulk-remediation/jobs`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(scope)
        });

        const job = await createResponse.json();
        this.currentJobId = job.id;

        // Show progress section
        document.getElementById('scope-section').style.display = 'none';
        document.getElementById('progress-section').style.display = 'block';

        // Start SSE
        this.eventSource = new EventSource(
            `/api/institutions/${this.institutionId}/bulk-remediation/jobs/${job.id}/run`
        );

        this.eventSource.onmessage = (event) => {
            const data = JSON.parse(event.data);
            this.handleProgressEvent(data);
        };

        this.eventSource.onerror = () => {
            this.eventSource.close();
        };
    },

    handleProgressEvent(data) {
        const progressBar = document.getElementById('progress-bar');
        const progressText = document.getElementById('progress-text');
        const progressList = document.getElementById('progress-list');

        switch (data.event) {
            case 'processing':
                progressBar.style.width = `${data.progress}%`;
                progressText.textContent = `${Math.round(data.progress)}%`;
                this.addProgressItem(data.document_name, 'running');
                break;

            case 'complete':
                this.updateProgressItem(data.document_name, 'complete', data.changes);
                break;

            case 'failed':
                this.updateProgressItem(data.document_name, 'failed', data.error);
                break;

            case 'done':
                this.eventSource.close();
                progressBar.style.width = '100%';
                progressText.textContent = '100%';
                this.showApprovalSection();
                break;
        }
    },

    addProgressItem(name, status) {
        const list = document.getElementById('progress-list');
        const item = document.createElement('div');
        item.className = `progress-item status-${status}`;
        item.id = `item-${name.replace(/\W/g, '')}`;
        item.innerHTML = `
            <span class="item-icon">${status === 'running' ? '⟳' : ''}</span>
            <span class="item-name">${name}</span>
            <span class="item-status">${status}</span>
        `;
        list.appendChild(item);
    },

    updateProgressItem(name, status, detail) {
        const itemId = `item-${name.replace(/\W/g, '')}`;
        const item = document.getElementById(itemId);
        if (item) {
            item.className = `progress-item status-${status}`;
            item.querySelector('.item-icon').textContent = status === 'complete' ? '✓' : '✗';
            item.querySelector('.item-status').textContent =
                status === 'complete' ? `${detail} fixes` : detail;
        }
    },

    async pauseJob() {
        await fetch(`/api/institutions/${this.institutionId}/bulk-remediation/jobs/${this.currentJobId}/pause`, {
            method: 'POST'
        });
        document.getElementById('pause-btn').textContent = 'Resume';
        document.getElementById('pause-btn').onclick = () => this.resumeJob();
    },

    async resumeJob() {
        await fetch(`/api/institutions/${this.institutionId}/bulk-remediation/jobs/${this.currentJobId}/resume`, {
            method: 'POST'
        });
        document.getElementById('pause-btn').textContent = 'Pause';
        document.getElementById('pause-btn').onclick = () => this.pauseJob();
    },

    cancelJob() {
        if (this.eventSource) {
            this.eventSource.close();
        }
        // Redirect back
        window.location.reload();
    },

    async showApprovalSection() {
        document.getElementById('progress-section').style.display = 'none';
        document.getElementById('approval-section').style.display = 'block';

        // Load job details
        const response = await fetch(
            `/api/institutions/${this.institutionId}/bulk-remediation/jobs/${this.currentJobId}`
        );
        const job = await response.json();

        const list = document.getElementById('approval-list');
        list.innerHTML = job.items.filter(i => i.status === 'complete').map(item => `
            <div class="approval-item" data-id="${item.id}">
                <input type="checkbox" class="approval-checkbox" checked>
                <div class="item-info">
                    <span class="item-name">${item.document_name}</span>
                    <span class="item-detail">${item.changes_count} changes | confidence: ${(item.confidence * 100).toFixed(0)}%</span>
                </div>
                <div class="item-actions">
                    <button class="btn btn-sm btn-secondary" onclick="BulkRemediation.viewDiff('${item.id}')">View Diff</button>
                    <button class="btn btn-sm btn-success" onclick="BulkRemediation.approveItem('${item.id}')">Approve</button>
                    <button class="btn btn-sm btn-danger" onclick="BulkRemediation.rejectItem('${item.id}')">Reject</button>
                </div>
            </div>
        `).join('');
    },

    async approveItem(itemId) {
        await fetch(`/api/institutions/${this.institutionId}/bulk-remediation/items/${itemId}/approve`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ approved_by: 'user' })
        });

        const item = document.querySelector(`[data-id="${itemId}"]`);
        item.classList.add('approved');
        item.querySelector('.btn-success').disabled = true;
    },

    async rejectItem(itemId) {
        await fetch(`/api/institutions/${this.institutionId}/bulk-remediation/items/${itemId}/reject`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ approved_by: 'user' })
        });

        const item = document.querySelector(`[data-id="${itemId}"]`);
        item.classList.add('rejected');
        item.querySelector('.btn-danger').disabled = true;
    },

    async approveAll() {
        await fetch(`/api/institutions/${this.institutionId}/bulk-remediation/jobs/${this.currentJobId}/approve-all`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ approved_by: 'user' })
        });

        // Update UI
        document.querySelectorAll('.approval-item').forEach(item => {
            item.classList.add('approved');
        });

        // Show success message
        alert('All remediations approved! Finals have been created.');
    },

    viewDiff(itemId) {
        // Open diff viewer modal or navigate to document workbench
        window.open(`/institutions/${this.institutionId}/workbench?item=${itemId}`, '_blank');
    }
};

// Initialize when page loads
document.addEventListener('DOMContentLoaded', () => {
    const institutionId = document.body.dataset.institutionId;
    BulkRemediation.init(institutionId);
});
```

## Flask Route

```python
@app.route('/institutions/<institution_id>/bulk-remediation')
def bulk_remediation_page(institution_id):
    institution = workspace_manager.load_institution(institution_id)
    if not institution:
        return redirect('/dashboard')

    return render_template('bulk_remediation.html',
        institution=institution,
        doc_types=get_doc_types(institution_id),
        programs=get_programs(institution_id),
        readiness_score=compute_readiness(institution_id)
    )
```
