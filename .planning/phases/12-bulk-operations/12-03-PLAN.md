       const batchOps = BatchOperations.init('{{ institution.id }}', 'audit');

       // Attach checkbox change handlers
       document.querySelectorAll('.batch-select-checkbox').forEach(cb => {
           cb.addEventListener('change', (e) => {
               const docId = e.target.dataset.docId;
               const docData = {
                   id: docId,
                   name: e.target.dataset.docName,
                   doc_type: e.target.dataset.docType
               };
               batchOps.selectionManager.toggle(docId, docData);
               batchOps.actionBar.updateCount(batchOps.selectionManager.getCount());
           });
       });

       // Select all handler
       document.getElementById('selectAll').addEventListener('change', (e) => {
           if (e.target.checked) {
               const allDocs = [...document.querySelectorAll('.batch-select-checkbox')]
                   .map(cb => ({id: cb.dataset.docId, name: cb.dataset.docName, doc_type: cb.dataset.docType}));
               batchOps.selectionManager.selectAll(allDocs);
           } else {
               batchOps.selectionManager.clearSelection();
           }
           batchOps.actionBar.updateCount(batchOps.selectionManager.getCount());
