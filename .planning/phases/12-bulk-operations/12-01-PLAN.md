   - metadata TEXT (JSON, for storing failed item IDs, retry info)

   - id TEXT PRIMARY KEY
   - batch_id TEXT NOT NULL (FK to batch_operations)
   - document_id TEXT NOT NULL
   - document_name TEXT (for display)
   - status TEXT DEFAULT 'pending' (pending, running, completed, failed)
   - task_id TEXT (TaskQueue task ID)
   - result_path TEXT (path to audit/remediation result file)
   - error TEXT (error message if failed)
   - input_tokens INTEGER DEFAULT 0
   - output_tokens INTEGER DEFAULT 0
   - duration_ms INTEGER DEFAULT 0
   - findings_count INTEGER DEFAULT 0 (for audit results)
   - created_at TEXT NOT NULL
   - started_at TEXT (nullable)
   - completed_at TEXT (nullable)

3. Indexes:
   - idx_batch_operations_institution ON batch_operations(institution_id)
   - idx_batch_operations_status ON batch_operations(status)
   - idx_batch_items_batch ON batch_items(batch_id)
