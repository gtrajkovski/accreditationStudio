## Base Package Implementation

File: `src/accreditors/base_package.py`

```python
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Dict, Any, List, Optional


@dataclass
class StandardsSource:
    """A source URL for standards documents."""
    url: str
    format: str  # pdf, html, docx
    name: str
    fetch_cadence: str = "monthly"


@dataclass
class StandardsTree:
    """Parsed standards hierarchy."""
    accreditor: str
    version: str
    parsed_at: str
    sections: List[Dict[str, Any]]
    total_standards: int


class AccreditorPackage(ABC):
    """Base class for accreditor packages."""

    @property
    @abstractmethod
    def manifest(self) -> Dict[str, Any]:
        """Return package manifest."""
        pass

    @abstractmethod
    def get_sources(self) -> List[StandardsSource]:
        """Return list of standards source URLs."""
        pass

    @abstractmethod
    def parse_standards(self, raw_path: str) -> StandardsTree:
        """Parse raw standards file into structured tree."""
        pass

    def get_crosswalk_seeds(self) -> Optional[Dict[str, Any]]:
        """Return optional crosswalk mappings to federal/state requirements."""
        return None

    def get_fetch_cadence(self) -> str:
        """Return fetch cadence: daily, weekly, monthly."""
        return self.manifest.get("fetch_cadence", "monthly")

    @property
    def code(self) -> str:
        """Return accreditor code."""
        return self.manifest["id"]

    @property
    def name(self) -> str:
        """Return accreditor full name."""
        return self.manifest["name"]
```

## Registry Implementation

File: `src/accreditors/registry.py`

```python
import os
import json
import importlib
from typing import Dict, List, Optional
from pathlib import Path

from src.accreditors.base_package import AccreditorPackage


class AccreditorRegistry:
    """Registry for discovering and accessing accreditor packages."""

    _packages: Dict[str, AccreditorPackage] = {}
    _manifests: Dict[str, Dict] = {}
    _discovered: bool = False

    @classmethod
    def discover(cls) -> None:
        """Discover all accreditor packages in src/accreditors/."""
        if cls._discovered:
            return

        accreditors_dir = Path(__file__).parent
        for subdir in accreditors_dir.iterdir():
            if not subdir.is_dir():
                continue
            if subdir.name.startswith("_") or subdir.name == "__pycache__":
                continue

            manifest_path = subdir / "manifest.json"
            if manifest_path.exists():
                with open(manifest_path) as f:
                    manifest = json.load(f)
                    cls._manifests[manifest["id"]] = manifest

        cls._discovered = True

    @classmethod
    def list_packages(cls) -> List[Dict]:
        """Return list of available accreditor packages."""
        cls.discover()
        return list(cls._manifests.values())

    @classmethod
    def get_package(cls, code: str) -> Optional[AccreditorPackage]:
        """Get instantiated package by code."""
        cls.discover()

        if code in cls._packages:
            return cls._packages[code]

        if code not in cls._manifests:
            return None

        try:
            module = importlib.import_module(f"src.accreditors.{code.lower()}")
            package_class = getattr(module, f"{code}Package", None)
            if package_class:
                cls._packages[code] = package_class()
                return cls._packages[code]
        except (ImportError, AttributeError):
            pass

        return None

    @classmethod
    def get_manifest(cls, code: str) -> Optional[Dict]:
        """Get manifest for accreditor without instantiating package."""
        cls.discover()
        return cls._manifests.get(code)
```

            self._manifest = json.load(f)

    def manifest(self) -> Dict[str, Any]:

        return [
            StandardsSource(

## API Blueprint

File: `src/api/accreditors.py`

```python
from flask import Blueprint, jsonify, request
from src.accreditors.registry import AccreditorRegistry

accreditors_bp = Blueprint("accreditors", __name__, url_prefix="/api/accreditors")


@accreditors_bp.route("", methods=["GET"])
def list_accreditors():
    """List all available accreditor packages."""
    packages = AccreditorRegistry.list_packages()
    return jsonify({"accreditors": packages})


@accreditors_bp.route("/<code>", methods=["GET"])
def get_accreditor(code: str):
    """Get accreditor package details."""
    manifest = AccreditorRegistry.get_manifest(code.upper())
    if not manifest:
        return jsonify({"error": "Accreditor not found"}), 404

    package = AccreditorRegistry.get_package(code.upper())
    sources = []
    if package:
        sources = [{"url": s.url, "format": s.format, "name": s.name} for s in package.get_sources()]

    return jsonify({
        "manifest": manifest,
        "sources": sources,
        "has_parser": package is not None,
        "has_crosswalk": package.get_crosswalk_seeds() is not None if package else False
    })


@accreditors_bp.route("/<code>/fetch", methods=["POST"])
def fetch_standards(code: str):
    """Trigger standards fetch for accreditor."""
    package = AccreditorRegistry.get_package(code.upper())
    if not package:
        return jsonify({"error": "Accreditor not found or no parser available"}), 404

    # Enqueue fetch task
    # This would integrate with standards harvester
    return jsonify({"status": "queued", "accreditor": code.upper()})
```

import pytest
from src.accreditors.registry import AccreditorRegistry


def test_registry_discovers_packages():
    """Verify all packages are discovered."""

    codes = [p["id"] for p in packages]
    assert "ACCSC" in codes
    assert len(packages) >= 5


def test_accsc_manifest_valid():

    manifest = AccreditorRegistry.get_manifest("ACCSC")
    assert manifest is not None
    assert manifest["id"] == "ACCSC"
    assert manifest["type"] == "institutional"
    assert "standards_url" in manifest

    """Verify ACCSC package returns sources."""
    package = AccreditorRegistry.get_package("ACCSC")
    assert package is not None
    sources = package.get_sources()
    assert len(sources) >= 1
    assert sources[0].format == "pdf"
