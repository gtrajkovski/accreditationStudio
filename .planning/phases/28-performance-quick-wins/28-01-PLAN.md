### Task 1: Add HTTP Cache Headers for Static Assets

1. Modify `add_security_headers()` to add cache headers for static files
2. Add versioning query param support for cache busting

**Implementation:**
```python
@app.after_request
def add_security_headers(response):
    """Add security and cache headers to all responses."""
    # Security headers
    response.headers["X-Content-Type-Options"] = "nosniff"
    response.headers["X-Frame-Options"] = "SAMEORIGIN"
    response.headers["X-XSS-Protection"] = "1; mode=block"
    response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"

    # Cache headers for static assets
    if request.path.startswith('/static/'):
        # Cache static assets for 1 year (immutable with versioning)
        response.headers["Cache-Control"] = "public, max-age=31536000, immutable"
    else:
        # No cache for dynamic content
        response.headers["Cache-Control"] = "no-store, must-revalidate"

    return response
```

### Task 3: Fix Portfolio N+1 Query with Batch Loading

**File:** `src/services/portfolio_service.py`

**Problem:** Lines 351-362 call `get_or_compute_readiness()` in a loop, causing 4 queries × N institutions.

**Solution:** Batch load recent snapshots for all institution IDs, then compute missing ones.

**Changes:**
1. Add `get_batch_readiness_snapshots()` function
2. Modify `compute_portfolio_readiness()` to use batch loading

**Implementation:**
```python
def get_batch_readiness_snapshots(
    institution_ids: List[str],
    cache_window_minutes: int = 10
) -> Dict[str, Dict[str, Any]]:
    """Batch load recent readiness snapshots for multiple institutions.

    Returns dict mapping institution_id -> readiness dict (or None if stale/missing).
    """
    if not institution_ids:
        return {}

    conn = get_conn()
    cutoff = (datetime.utcnow() - timedelta(minutes=cache_window_minutes)).isoformat()

    # Single query for all institutions
    placeholders = ",".join("?" * len(institution_ids))
    cursor = conn.execute(f"""
        SELECT institution_id, total_score, documents_score, compliance_score,
               evidence_score, consistency_score, computed_at
        FROM readiness_snapshots
        WHERE institution_id IN ({placeholders})
          AND computed_at > ?
        ORDER BY computed_at DESC
    """, (*institution_ids, cutoff))

    results = {}
    seen = set()
    for row in cursor:
        inst_id = row["institution_id"]
        if inst_id not in seen:  # Take most recent per institution
            seen.add(inst_id)
            results[inst_id] = {
                "total": row["total_score"],
                "documents": row["documents_score"],
                "compliance": row["compliance_score"],
                "evidence": row["evidence_score"],
                "consistency": row["consistency_score"],
            }

    return results
```

CREATE INDEX IF NOT EXISTS idx_audit_findings_run_status

ON document_versions(document_id, version_type);

CREATE INDEX IF NOT EXISTS idx_readiness_snapshots_inst_time
ON readiness_snapshots(institution_id, computed_at DESC);

-- Evidence refs: often joined with findings
CREATE INDEX IF NOT EXISTS idx_evidence_refs_finding
ON evidence_refs(finding_id);

-- Agent sessions: often filtered by institution + status
CREATE INDEX IF NOT EXISTS idx_agent_sessions_inst_status
ON agent_sessions(institution_id, status);
