CREATE TABLE IF NOT EXISTS user_invitations (
    id TEXT PRIMARY KEY,
    email TEXT NOT NULL,
    role TEXT NOT NULL DEFAULT 'viewer',
    institution_id TEXT,
    invited_by TEXT NOT NULL,
    token TEXT UNIQUE NOT NULL,

    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    FOREIGN KEY (institution_id) REFERENCES institutions(id),
    FOREIGN KEY (invited_by) REFERENCES users(id)

CREATE TABLE IF NOT EXISTS user_permissions (
    id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL,
    institution_id TEXT NOT NULL,
    permission TEXT NOT NULL,
    granted_by TEXT,
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (institution_id) REFERENCES institutions(id),
    FOREIGN KEY (granted_by) REFERENCES users(id),
    UNIQUE(user_id, institution_id, permission)
);

CREATE INDEX IF NOT EXISTS idx_invitations_token ON user_invitations(token);
CREATE INDEX IF NOT EXISTS idx_invitations_email ON user_invitations(email);
CREATE INDEX IF NOT EXISTS idx_permissions_user ON user_permissions(user_id);
CREATE INDEX IF NOT EXISTS idx_permissions_institution ON user_permissions(institution_id);

### Task 5: Role Guard Decorators

**File:** `app.py` (add decorators)

```python
ROLE_HIERARCHY = ['viewer', 'department_head', 'compliance_officer', 'admin', 'owner']

def require_role(*allowed_roles):
    def decorator(f):
        @wraps(f)
        def decorated(*args, **kwargs):
            if not app.config.get('AUTH_ENABLED', True):
                return f(*args, **kwargs)
            user = g.get('current_user')
            if not user or user.get('role') not in allowed_roles:
                return jsonify({'error': 'Insufficient permissions'}), 403
            return f(*args, **kwargs)
        return decorated
    return decorator

def require_minimum_role(minimum_role):
    min_index = ROLE_HIERARCHY.index(minimum_role)
    allowed = set(ROLE_HIERARCHY[min_index:])
    return require_role(*allowed)
```
