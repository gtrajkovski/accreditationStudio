```python
def _compute_file_hash(file_path: str) -> Optional[str]:
    """Compute SHA256 hash of a file."""
    try:
        path = Path(file_path)
        if not path.exists():
            return None
        sha256 = hashlib.sha256()
        with open(path, 'rb') as f:
            for chunk in iter(lambda: f.read(8192), b''):
                sha256.update(chunk)
        return sha256.hexdigest()
    except Exception as e:
        logger.warning(f"Failed to compute hash for {file_path}: {e}")
        return None
```

```python
existing_doc_id = request.form.get('document_id')  # If replacing existing

    # Get existing document from database
    conn = get_conn()
    cursor = conn.execute(
        "SELECT file_sha256, institution_id FROM documents WHERE id = ?",
        (existing_doc_id,)

        old_hash = existing["file_sha256"]

            # Document changed - store previous text for diff (per D-12)
```
